    Anotações para construção do módulo 2:

    Exemplo simples de declaração de classe e criação de objeto # 

    #include <iostream>
using namespace std;

class Pessoa {
    private:
        string nome;

    public:
        void setNome(string n){
            nome = n;
        }
        string getNome(){
            return nome;
        }
        void apresentacao(){
            cout << "Olá, meu nome é " << nome << "." << endl;
        }
        
};

int main() {

    Pessoa p1, p2, p3;
    p1.apresentacao();
    
    p2.setNome("Maria");
    p2.apresentacao();

    p3.setNome("João");
    p3.apresentacao();

    cout << "Pessoas cadastradas: " << p2.getNome() << " e " << p3.getNome() << "." << endl;

    return 0;
}

    A saída será:

    Olá, meu nome é .
Olá, meu nome é Maria.
Olá, meu nome é João.
Pessoas cadastradas: Maria e João.

    • Exemplo com mais parâmetros:
    #include <iostream>
using namespace std;

class Pessoa {
    private:
        string nome;
        int idade;

    public:
        //Construtor default      
        Pessoa (){
            nome = "(Não definido)";
            idade = -1;
        }
        //Construtor com parâmetro
        Pessoa (string n){
            nome = n;
        }
        //Construtor com parâmetro
        Pessoa (string n, int i){
            nome = n;
            idade = i;
        }
        void setNome(string n){
            nome = n;
        }
        string getNome(){
            return nome;
        }
        void setIdade(int i){
            idade = i;
        }
        int getIdade(){
            return idade;
        }
        void apresentacao(){
            cout << "Olá, meu nome é " << nome << ". Tenho " << idade <<" anos." << endl;
        }
};

int main() {

    Pessoa p1("Ana", 21), p2 ("Maria"), p3;
    p1.apresentacao();

    p2.setIdade(22);
    p2.apresentacao();

    p3.setNome("João");
    p3.apresentacao();

    return 0;
}

    A saída será:

    Olá, meu nome é Ana. Tenho 21 anos.
Olá, meu nome é Maria. Tenho 22 anos.
Olá, meu nome é João. Tenho -1 anos.

    Construtor com lista de inicialização de membros #

using namespace std;

    p1.apresentacao();

    p2.setIdade(22);
    p2.apresentacao();

    p3.setNome("João");
    p3.apresentacao();

    Olá, meu nome é Ana. Tenho 21 anos.

    Definição de classe em arquivos separados #

    PASSO 1: Arquivo de cabeçalho #

    Arquivo Pessoa.hpp

    #ifndef PESSOA_HPP
#define PESSOA_HPP

#include <string>
using namespace std;

class Pessoa{
private:
    string nome;
    int idade;

public:
    Pessoa (string n, int i);

    void setNome(string n);
    string getNome();
    void setIdade(int i);
    int getIdade();
    void apresentacao();

};

#endif // PESSOA_HPP

    PASSO 2: Arquivo de implementação #

    Arquivo Pessoa.cpp

    #include "Pessoa.hpp"
#include <iostream>

Pessoa::Pessoa(string n, int i): nome(n), idade(i){}

void Pessoa::setNome(string n){
    nome = n;
}
string Pessoa::getNome(){
    return nome;
}
void Pessoa::setIdade(int i){
    idade = i;
}
int Pessoa::getIdade(){
    return idade;
}
void Pessoa::apresentacao(){
    cout << "Olá, meu nome é " << nome << ". Tenho " << idade <<" anos." << endl;
}

}

    Herança #

    Uma nova classe que reaproveita os atributos e métodos de uma classe existente.

    PASSO 1: Arquivo de cabeçalho #

    Arquivo Aluno.hpp

    #ifndef ALUNO_HPP
#define ALUNO_HPP

#include "Pessoa.hpp"
#include <string>

using namespace  std;

class Aluno : public Pessoa{ // Herança

    private:
        string matricula;

    public:
        Aluno(string nome, int idade, string mat);

        string getMatricula();
        void setMatricula(string m);

        void apresentacao(); //sobrescrita

};

#endif // ALUNO_HPP

    PASSO 2: Arquivo de implementação #

    Arquivo Aluno.cpp

    #include "Aluno.hpp"
#include <iostream>

using namespace std;

Aluno::Aluno(string n, int i, string mat): Pessoa(n, i), matricula(mat){}

string Aluno::getMatricula(){
    return matricula;
}

void Aluno::setMatricula(string m){
    matricula = m;
}

void Aluno::apresentacao(){
    cout << "[Aluno]" << endl;
    cout << " Nome: " << Aluno::getNome() << endl;
    cout << " Idade: " << Aluno::getIdade() << endl;
    cout << " Matricula: " << getMatricula() << endl;
}
